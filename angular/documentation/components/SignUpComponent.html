import {
  Component,
  OnInit,
  Input,
  ElementRef,

} from "@angular/core";
import { IUser, emptyUser } from "@/user/interfaces/iuser";
import { NgModel } from '@angular/forms';
import { cleanSession } from 'selenium-webdriver/safari';
@Component({
  // tslint:disable-next-line:component-selector
  selector: "sh-sign-up",
  templateUrl: "./sign-up.component.html",
  styleUrls: ["./sign-up.component.scss"]
})
export class SignUpComponent implements OnInit {
  user: IUser = emptyUser();
  userTypes: string[] = ["Admin", "Bussines Owner", "Client"];
  @Output() signUp = new EventEmitter<IUser>();
  eyeStatus = "password";
  defaultType = "Client";
  verEyeStatus = "password";

  constructor() { }

  ngOnInit() {
    this.user.type = "Client";
  }

  comparePasswords(firstPassword: string, secondPassword: string) {
    return firstPassword === secondPassword;
  }
  onSubmit() {
    this.signUp.emit(this.user);
    alert("you were signed-up successfuly");

  }
